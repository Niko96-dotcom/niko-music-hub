```swift
import Foundation

/// 4-category error taxonomy for Outside Cubase Hub.
/// Every error in the app falls into one of these categories.
public enum AppErrorCategory: String, CaseIterable {
    case permission        // "Permission Required" / "Recording Restricted"
    case helperTool        // "[Tool] Missing / Error"
    case conversionFile    // "Could Not [Verb]" / "[Noun] Failed"
    case inputURL          // "Invalid [Noun]" / "[Noun] Not Supported"
}

/// Standard error card content structure used by all tools.
public struct AppErrorCard {
    public let category: AppErrorCategory
    public let label: String            // e.g. "Could Not Save Recording"
    public let icon: String             // SF Symbol name, e.g. "externaldrive.badge.xmark"
    public let body: String             // Plain-language explanation
    public let recoveryActions: [RecoveryAction]

    public struct RecoveryAction {
        public let label: String       // Button label
        public let style: ActionStyle   // primary, secondary, destructive
        public enum ActionStyle { case primary, secondary, destructive }
        public let action: RecoveryActionType
    }

    public enum RecoveryActionType {
        case openSystemSettings
        case tryAgain
        case chooseToolPath
        case revealInFinder
        case openTerminal
        case dismiss
    }
}
```

Also create `Sources/AppCore/Errors/StandardErrorCard.swift` — a reusable SwiftUI view that renders `AppErrorCard`:
- Label: `.system(size: 14, weight: .semibold)` colored by category (permission=orange, helperTool=red, conversionFile=red, inputURL=orange)
- Body: `.system(size: 12)` in `.secondary` foreground
- Recovery buttons: `.borderedProminent` for primary, `.bordered` for secondary, `.bordered` + `.red` for destructive
- Icon: `Image(systemName: card.icon)` displayed alongside the label
